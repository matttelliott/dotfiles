=====================
@link w/ URI
=====================
/**
 * @link http://example.com
 * @link https://example.com
 * @link http://www.example.com
 * @link https://www.example.com
 */
---
(document
  (tag (tag_name) (uri))
  (tag (tag_name) (uri))
  (tag (tag_name) (uri))
  (tag (tag_name) (uri)))

=====================
@see w/ FQSEN for classnames
=====================
/**
 * @see \MyClass
 * @see \Vendor\Package\ClassName
 * @see MyClass
 * @see Vendor\Package\ClassName
 */
---
(document
  (tag
    (tag_name)
    (fqsen
      (named_type
        (qualified_name
          (namespace_name_as_prefix)
          (name)))))
  (tag
    (tag_name)
    (fqsen
      (named_type
        (qualified_name
          (namespace_name_as_prefix
            (namespace_name (name) (name)))
          (name)))))
  (tag
    (tag_name)
    (fqsen
      (named_type (name))))
  (tag
    (tag_name)
    (fqsen
      (named_type
        (qualified_name
          (namespace_name_as_prefix
            (namespace_name (name) (name)))
          (name))))))
